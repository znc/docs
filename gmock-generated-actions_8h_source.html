00784 // The ACTION* family of macros can be used in a namespace scope to
00785 // define custom actions easily.  The syntax:
00786 //
00787 //   ACTION(name) { statements; }
00788 //
00789 // will define an action with the given name that executes the
00790 // statements.  The value returned by the statements will be used as
00791 // the return value of the action.  Inside the statements, you can
00792 // refer to the K-th (0-based) argument of the mock function by
00793 // 'argK', and refer to its type by 'argK_type'.  For example:
00794 //
00795 //   ACTION(IncrementArg1) {
00796 //     arg1_type temp = arg1;
00797 //     return ++(*temp);
00798 //   }
00799 //
00800 // allows you to write
00801 //
00802 //   ...WillOnce(IncrementArg1());
00803 //
00804 // You can also refer to the entire argument tuple and its type by
00805 // 'args' and 'args_type', and refer to the mock function type and its
00806 // return type by 'function_type' and 'return_type'.
00807 //
00808 // Note that you don't need to specify the types of the mock function
00809 // arguments.  However rest assured that your code is still type-safe:
00810 // you'll get a compiler error if *arg1 doesn't support the ++
00811 // operator, or if the type of ++(*arg1) isn't compatible with the
00812 // mock function's return type, for example.
00813 //
00814 // Sometimes you'll want to parameterize the action.   For that you can use
00815 // another macro:
00816 //
00817 //   ACTION_P(name, param_name) { statements; }
00818 //
00819 // For example:
00820 //
00821 //   ACTION_P(Add, n) { return arg0 + n; }
00822 //
00823 // will allow you to write:
00824 //
00825 //   ...WillOnce(Add(5));
00826 //
00827 // Note that you don't need to provide the type of the parameter
00828 // either.  If you need to reference the type of a parameter named
00829 // 'foo', you can write 'foo_type'.  For example, in the body of
00830 // ACTION_P(Add, n) above, you can write 'n_type' to refer to the type
00831 // of 'n'.
00832 //
00833 // We also provide ACTION_P2, ACTION_P3, ..., up to ACTION_P10 to support
00834 // multi-parameter actions.
00835 //
00836 // For the purpose of typing, you can view
00837 //
00838 //   ACTION_Pk(Foo, p1, ..., pk) { ... }
00839 //
00840 // as shorthand for
00841 //
00842 //   template <typename p1_type, ..., typename pk_type>
00843 //   FooActionPk<p1_type, ..., pk_type> Foo(p1_type p1, ..., pk_type pk) { ... }
00844 //
00845 // In particular, you can provide the template type arguments
00846 // explicitly when invoking Foo(), as in Foo<long, bool>(5, false);
00847 // although usually you can rely on the compiler to infer the types
00848 // for you automatically.  You can assign the result of expression
00849 // Foo(p1, ..., pk) to a variable of type FooActionPk<p1_type, ...,
00850 // pk_type>.  This can be useful when composing actions.
00851 //
00852 // You can also overload actions with different numbers of parameters:
00853 //
00854 //   ACTION_P(Plus, a) { ... }
00855 //   ACTION_P2(Plus, a, b) { ... }
00856 //
00857 // While it's tempting to always use the ACTION* macros when defining
00858 // a new action, you should also consider implementing ActionInterface
00859 // or using MakePolymorphicAction() instead, especially if you need to
00860 // use the action a lot.  While these approaches require more work,
00861 // they give you more control on the types of the mock function
00862 // arguments and the action parameters, which in general leads to
00863 // better compiler error messages that pay off in the long run.  They
00864 // also allow overloading actions based on parameter types (as opposed
00865 // to just based on the number of parameters).
00866 //
00867 // CAVEAT:
00868 //
00869 // ACTION*() can only be used in a namespace scope.  The reason is
00870 // that C++ doesn't yet allow function-local types to be used to
00871 // instantiate templates.  The up-coming C++0x standard will fix this.
00872 // Once that's done, we'll consider supporting using ACTION*() inside
00873 // a function.
00874 //
00875 // MORE INFORMATION:
00876 //
00877 // To learn more about using these macros, please search for 'ACTION'
00878 // on http://code.google.com/p/googlemock/wiki/CookBook.

00894 // Sometimes you want to give an action explicit template parameters
00895 // that cannot be inferred from its value parameters.  ACTION() and
00896 // ACTION_P*() don't support that.  ACTION_TEMPLATE() remedies that
00897 // and can be viewed as an extension to ACTION() and ACTION_P*().
00898 //
00899 // The syntax:
00900 //
00901 //   ACTION_TEMPLATE(ActionName,
00902 //                   HAS_m_TEMPLATE_PARAMS(kind1, name1, ..., kind_m, name_m),
00903 //                   AND_n_VALUE_PARAMS(p1, ..., p_n)) { statements; }
00904 //
00905 // defines an action template that takes m explicit template
00906 // parameters and n value parameters.  name_i is the name of the i-th
00907 // template parameter, and kind_i specifies whether it's a typename,
00908 // an integral constant, or a template.  p_i is the name of the i-th
00909 // value parameter.
00910 //
00911 // Example:
00912 //
00913 //   // DuplicateArg<k, T>(output) converts the k-th argument of the mock
00914 //   // function to type T and copies it to *output.
00915 //   ACTION_TEMPLATE(DuplicateArg,
00916 //                   HAS_2_TEMPLATE_PARAMS(int, k, typename, T),
00917 //                   AND_1_VALUE_PARAMS(output)) {
00918 //     *output = T(::testing::get<k>(args));
00919 //   }
00920 //   ...
00921 //     int n;
00922 //     EXPECT_CALL(mock, Foo(_, _))
00923 //         .WillOnce(DuplicateArg<1, unsigned char>(&n));
00924 //
00925 // To create an instance of an action template, write:
00926 //
00927 //   ActionName<t1, ..., t_m>(v1, ..., v_n)
00928 //
00929 // where the ts are the template arguments and the vs are the value
00930 // arguments.  The value argument types are inferred by the compiler.
00931 // If you want to explicitly specify the value argument types, you can
00932 // provide additional template arguments:
00933 //
00934 //   ActionName<t1, ..., t_m, u1, ..., u_k>(v1, ..., v_n)
00935 //
00936 // where u_i is the desired type of v_i.
00937 //
00938 // ACTION_TEMPLATE and ACTION/ACTION_P* can be overloaded on the
00939 // number of value parameters, but not on the number of template
00940 // parameters.  Without the restriction, the meaning of the following
00941 // is unclear:
00942 //
00943 //   OverloadedAction<int, bool>(x);
00944 //
00945 // Are we using a single-template-parameter action where 'bool' refers
00946 // to the type of x, or are we using a two-template-parameter action
00947 // where the compiler is asked to infer the type of x?
00948 //
00949 // Implementation notes:
00950 //
00951 // GMOCK_INTERNAL_*_HAS_m_TEMPLATE_PARAMS and
00952 // GMOCK_INTERNAL_*_AND_n_VALUE_PARAMS are internal macros for
00953 // implementing ACTION_TEMPLATE.  The main trick we use is to create
00954 // new macro invocations when expanding a macro.  For example, we have
00955 //
00956 //   #define ACTION_TEMPLATE(name, template_params, value_params)
00957 //       ... GMOCK_INTERNAL_DECL_##template_params ...
00958 //
00959 // which causes ACTION_TEMPLATE(..., HAS_1_TEMPLATE_PARAMS(typename, T), ...)
00960 // to expand to
00961 //
00962 //       ... GMOCK_INTERNAL_DECL_HAS_1_TEMPLATE_PARAMS(typename, T) ...
00963 //
00964 // Since GMOCK_INTERNAL_DECL_HAS_1_TEMPLATE_PARAMS is a macro, the
00965 // preprocessor will continue to expand it to
00966 //
00967 //       ... typename T ...
00968 //
00969 // This technique conforms to the C++ standard and is portable.  It
00970 // allows us to implement action templates using O(N) code, where N is
00971 // the maximum number of template/value parameters supported.  Without
00972 // using it, we'd have to devote O(N^2) amount of code to implement all
00973 // combinations of m and n.

00978     name1) kind0 name0, kind1 name1

02094 // Various overloads for InvokeArgument<N>().
02095 //
02096 // The InvokeArgument<N>(a1, a2, ..., a_k) action invokes the N-th
02097 // (0-based) argument, which must be a k-ary callable, of the mock
02098 // function, with arguments a1, a2, ..., a_k.
02099 //
02100 // Notes:
02101 //
02102 //   1. The arguments are passed by value by default.  If you need to
02103 //   pass an argument by reference, wrap it inside ByRef().  For
02104 //   example,
02105 //
02106 //     InvokeArgument<1>(5, string("Hello"), ByRef(foo))
02107 //
02108 //   passes 5 and string("Hello") by value, and passes foo by
02109 //   reference.
02110 //
02111 //   2. If the callable takes an argument by reference but ByRef() is
02112 //   not used, it will receive the reference to a copy of the value,
02113 //   instead of the original value.  For example, when the 0-th
02114 //   argument of the mock function takes a const string&, the action
02115 //
02116 //     InvokeArgument<0>(string("Hello"))
02117 //
02118 //   makes a copy of the temporary string("Hello") object and passes a
02119 //   reference of the copy, instead of the original temporary object,
02120 //   to the callable.  This makes it easy for a user to define an
02121 //   InvokeArgument action from temporary values and have it performed
02122 //   later.
