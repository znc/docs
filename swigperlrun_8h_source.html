00526 /*
00527   Search for a swig_type_info structure only by mangled name
00528   Search is a O(log #types)
00529 
00530   We start searching at module start, and finish searching when start == end.
00531   Note: if start == end at the beginning of the function, we go all the way around
00532   the circular list.
00533 */
00534 SWIGRUNTIME swig_type_info *
00535 SWIG_MangledTypeQueryModule(swig_module_info *start,
00536                             swig_module_info *end,
00537                             const char *name) {
00538   swig_module_info *iter = start;
00539   do {
00540     if (iter->size) {
00541       size_t l = 0;
00542       size_t r = iter->size - 1;
00543       do {
00544         /* since l+r >= 0, we can (>> 1) instead (/ 2) */
00545         size_t i = (l + r) >> 1;
00546         const char *iname = iter->types[i]->name;
00547         if (iname) {
00548           int compare = strcmp(name, iname);
00549           if (compare == 0) {
00550             return iter->types[i];
00551           } else if (compare < 0) {
00552             if (i) {
00553               r = i - 1;
00554             } else {
00555               break;
00556             }
00557           } else if (compare > 0) {
00558             l = i + 1;
00559           }
00560         } else {
00561           break; /* should never happen */
00562         }
00563       } while (l <= r);
00564     }
00565     iter = iter->next;
00566   } while (iter != end);
00567   return 0;

00623 */
00624 SWIGRUNTIME const char *
00625 SWIG_UnpackData(const char *c, void *ptr, size_t sz) {
00626   unsigned char *u = (unsigned char *) ptr;
00627   const unsigned char *eu = u + sz;
00628   for (; u != eu; ++u) {
00629     char d = *(c++);
00630     unsigned char uu;
00631     if ((d >= '0') && (d <= '9'))
00632       uu = ((d - '0') << 4);
00633     else if ((d >= 'a') && (d <= 'f'))
00634       uu = ((d - ('a'-10)) << 4);
00635     else
00636       return (char *) 0;
00637     d = *(c++);
00638     if ((d >= '0') && (d <= '9'))
00639       uu |= (d - '0');
00640     else if ((d >= 'a') && (d <= 'f'))
00641       uu |= (d - ('a'-10));
00642     else
00643       return (char *) 0;
00644     *u = uu;
00645   }
00646   return c;
00647 }

01049 /* Identical to SWIG_TypeCheck, except for strcmp comparison */
01050 SWIGRUNTIME swig_cast_info *
01051 SWIG_TypeProxyCheck(const char *c, swig_type_info *ty) {
01052   if (ty) {
01053     swig_cast_info *iter = ty->cast;
01054     while (iter) {
01055       if (strcmp(SWIG_Perl_TypeProxyName(iter->type), c) == 0) {
01056         if (iter == ty->cast)
01057           return iter;
01058         /* Move iter to the top of the linked list */
01059         iter->prev->next = iter->next;
01060         if (iter->next)
01061           iter->next->prev = iter->prev;
01062         iter->next = ty->cast;
01063         iter->prev = 0;
01064         if (ty->cast) ty->cast->prev = iter;
01065         ty->cast = iter;
01066         return iter;
01067       }
01068       iter = iter->next;
01069     }
01070   }
01071   return 0;
01072 }

01076 SWIGRUNTIME int
01077 SWIG_Perl_ConvertPtrAndOwn(SWIG_MAYBE_PERL_OBJECT SV *sv, void **ptr, swig_type_info *_t, int flags, int *own) {
01078   swig_cast_info *tc;
01079   void *voidptr = (void *)0;
01080   SV *tsv = 0;
01081 
01082   if (own)
01083     *own = 0;
01084 
01085   /* If magical, apply more magic */
01086   if (SvGMAGICAL(sv))
01087     mg_get(sv);
01088 
01089   /* Check to see if this is an object */
01090   if (sv_isobject(sv)) {
01091     IV tmp = 0;
01092     tsv = (SV*) SvRV(sv);
01093     if ((SvTYPE(tsv) == SVt_PVHV)) {
01094       MAGIC *mg;
01095       if (SvMAGICAL(tsv)) {
01096         mg = mg_find(tsv,'P');
01097         if (mg) {
01098           sv = mg->mg_obj;
01099           if (sv_isobject(sv)) {
01100             tsv = (SV*)SvRV(sv);
01101             tmp = SvIV(tsv);
01102           }
01103         }
01104       } else {
01105         return SWIG_ERROR;
01106       }
01107     } else {
01108       tmp = SvIV(tsv);
01109     }
01110     voidptr = INT2PTR(void *,tmp);
01111   } else if (! SvOK(sv)) {            /* Check for undef */
01112     *(ptr) = (void *) 0;
01113     return SWIG_OK;
01114   } else if (SvTYPE(sv) == SVt_RV) {  /* Check for NULL pointer */
01115     if (!SvROK(sv)) {
01116       /* In Perl 5.12 and later, SVt_RV == SVt_IV, so sv could be a valid integer value.  */
01117       if (SvIOK(sv)) {
01118         return SWIG_ERROR;
01119       } else {
01120         /* NULL pointer (reference to undef). */
01121         *(ptr) = (void *) 0;
01122         return SWIG_OK;
01123       }
01124     } else {
01125       return SWIG_ERROR;
01126     }
01127   } else {                            /* Don't know what it is */
01128     return SWIG_ERROR;
01129   }
01130   if (_t) {
01131     /* Now see if the types match */
01132     char *_c = HvNAME(SvSTASH(SvRV(sv)));
01133     tc = SWIG_TypeProxyCheck(_c,_t);
01134 #ifdef SWIG_DIRECTORS
01135     if (!tc && !sv_derived_from(sv,SWIG_Perl_TypeProxyName(_t))) {
01136 #else
01137     if (!tc) {
01138 #endif
01139       return SWIG_ERROR;
01140     }
01141     {
01142       int newmemory = 0;
01143       *ptr = SWIG_TypeCast(tc,voidptr,&newmemory);
01144       if (newmemory == SWIG_CAST_NEW_MEMORY) {
01145         assert(own); /* badly formed typemap which will lead to a memory leak - it must set and use own to delete *ptr */
01146         if (own)
01147           *own = *own | SWIG_CAST_NEW_MEMORY;
01148       }
01149     }
01150   } else {
01151     *ptr = voidptr;
01152   }
01153 
01154   /* 
01155    *  DISOWN implementation: we need a perl guru to check this one.
01156    */
01157   if (tsv && (flags & SWIG_POINTER_DISOWN)) {
01158     /* 
01159      *  almost copy paste code from below SWIG_POINTER_OWN setting
01160      */
01161     SV *obj = sv;
01162     HV *stash = SvSTASH(SvRV(obj));
01163     GV *gv = *(GV**)hv_fetch(stash, "OWNER", 5, TRUE);
01164     if (isGV(gv)) {
01165       HV *hv = GvHVn(gv);
01166       /*
01167        * To set ownership (see below), a newSViv(1) entry is added. 
01168        * Hence, to remove ownership, we delete the entry.
01169        */
01170       if (hv_exists_ent(hv, obj, 0)) {
01171         hv_delete_ent(hv, obj, 0, 0);
01172       }
01173     }
01174   }
01175   return SWIG_OK;
01176 }
