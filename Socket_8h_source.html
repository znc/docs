89 CSListener L(iPort, sBindHost);

92 L.SetIsSSL(bSSL);
93 L.SetTimeout(iTimeout);

96#ifdef HAVE_IPV6

103 break;
104 case ADDR_ALL:
105 L.SetAFRequire(CSSockAddr::RAF_ANY);
106 break;
107 }
108#endif
109
110 return Listen(L, pcSock);
111 }
