00623 */
00624 SWIGRUNTIME const char *
00625 SWIG_UnpackData(const char *c, void *ptr, size_t sz) {
00626   unsigned char *u = (unsigned char *) ptr;
00627   const unsigned char *eu = u + sz;
00628   for (; u != eu; ++u) {
00629     char d = *(c++);
00630     unsigned char uu;
00631     if ((d >= '0') && (d <= '9'))
00632       uu = ((d - '0') << 4);
00633     else if ((d >= 'a') && (d <= 'f'))
00634       uu = ((d - ('a'-10)) << 4);
00635     else
00636       return (char *) 0;
00637     d = *(c++);
00638     if ((d >= '0') && (d <= '9'))
00639       uu |= (d - '0');
00640     else if ((d >= 'a') && (d <= 'f'))
00641       uu |= (d - ('a'-10));
00642     else
00643       return (char *) 0;
00644     *u = uu;
00645   }
00646   return c;
00647 }

01296     if (l < min) {

01298                    name, (min == max ? "" : "at least "), (int)min, (int)l);
01299       return 0;

01307         objs[i] = PyTuple_GET_ITEM(args, i);
01308       }
01309       for (; l < max; ++l) {
01310         objs[l] = 0;
01311       }
01312       return i + 1;
