322  void AddItem(const K& Item, const V& Val, unsigned int uTTL) {
323  if (!uTTL) {
324  // If time-to-live is zero we don't want to waste our time adding
325  // it
326  RemItem(Item); // Remove the item incase it already exists
327  return;
328  }
329 
330  m_mItems[Item] = value(CUtils::GetMillTime() + uTTL, Val);
331  }
332 
338  bool HasItem(const K& Item) {
339  Cleanup();
340  return (m_mItems.find(Item) != m_mItems.end());
341  }
342 
348  V* GetItem(const K& Item) {
349  Cleanup();
350  iterator it = m_mItems.find(Item);
351  if (it == m_mItems.end()) return nullptr;
352  return &it->second.second;
353  }
354 
360  bool RemItem(const K& Item) { return (m_mItems.erase(Item) != 0); }
361 
365  void Cleanup() {
366  iterator it = m_mItems.begin();
367 
368  while (it != m_mItems.end()) {
369  if (CUtils::GetMillTime() > (it->second.first)) {
370  m_mItems.erase(it++);
371  } else {
372  ++it;
373  }
374  }
375  }
376 
380  void Clear() { m_mItems.clear(); }
381 
385  std::map<K, V> GetItems() {
386  Cleanup();
387  std::map<K, V> mItems;
388  for (const auto& it : m_mItems) {
389  mItems[it.first] = it.second.second;
390  }
391  return mItems;
392  }
