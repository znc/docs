00826 // A MockFunction<F> class has one mock method whose type is F.  It is
00827 // useful when you just want your test code to emit some messages and
00828 // have Google Mock verify the right messages are sent (and perhaps at
00829 // the right times).  For example, if you are exercising code:
00830 //
00831 //   Foo(1);
00832 //   Foo(2);
00833 //   Foo(3);
00834 //
00835 // and want to verify that Foo(1) and Foo(3) both invoke
00836 // mock.Bar("a"), but Foo(2) doesn't invoke anything, you can write:
00837 //
00838 // TEST(FooTest, InvokesBarCorrectly) {
00839 //   MyMock mock;
00840 //   MockFunction<void(string check_point_name)> check;
00841 //   {
00842 //     InSequence s;
00843 //
00844 //     EXPECT_CALL(mock, Bar("a"));
00845 //     EXPECT_CALL(check, Call("1"));
00846 //     EXPECT_CALL(check, Call("2"));
00847 //     EXPECT_CALL(mock, Bar("a"));
00848 //   }
00849 //   Foo(1);
00850 //   check.Call("1");
00851 //   Foo(2);
00852 //   check.Call("2");
00853 //   Foo(3);
00854 // }
00855 //
00856 // The expectation spec says that the first Bar("a") must happen
00857 // before check point "1", the second Bar("a") must happen after check
00858 // point "2", and nothing should happen between the two check
00859 // points. The explicit check points make it easy to tell which
00860 // Bar("a") is called by which call to Foo().
00861 //
00862 // MockFunction<F> can also be used to exercise code that accepts
00863 // std::function<F> callbacks. To do so, use AsStdFunction() method
00864 // to create std::function proxy forwarding to original object's Call.
00865 // Example:
00866 //
00867 // TEST(FooTest, RunsCallbackWithBarArgument) {
00868 //   MockFunction<int(string)> callback;
00869 //   EXPECT_CALL(callback, Call("bar")).WillOnce(Return(1));
00870 //   Foo(callback.AsStdFunction());
00871 // }
