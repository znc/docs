00090 // Interface for iterating over elements provided by an implementation
00091 // of ParamGeneratorInterface<T>.
00092 template <typename T>
00093 class ParamIteratorInterface {
00094  public:
00095   virtual ~ParamIteratorInterface() {}
00096   // A pointer to the base generator instance.
00097   // Used only for the purposes of iterator comparison
00098   // to make sure that two iterators belong to the same generator.
00099   virtual const ParamGeneratorInterface<T>* BaseGenerator() const = 0;
00100   // Advances iterator to point to the next element
00101   // provided by the generator. The caller is responsible
00102   // for not calling Advance() on an iterator equal to
00103   // BaseGenerator()->End().
00104   virtual void Advance() = 0;
00105   // Clones the iterator object. Used for implementing copy semantics
00106   // of ParamIterator<T>.
00107   virtual ParamIteratorInterface* Clone() const = 0;
00108   // Dereferences the current iterator and provides (read-only) access
00109   // to the pointed value. It is the caller's responsibility not to call
00110   // Current() on an iterator equal to BaseGenerator()->End().
00111   // Used for implementing ParamGenerator<T>::operator*().
00112   virtual const T* Current() const = 0;
00113   // Determines whether the given iterator and other point to the same
00114   // element in the sequence generated by the generator.
00115   // Used for implementing ParamGenerator<T>::operator==().
00116   virtual bool Equals(const ParamIteratorInterface& other) const = 0;
00117 };

00119 // Class iterating over elements provided by an implementation of
00120 // ParamGeneratorInterface<T>. It wraps ParamIteratorInterface<T>
00121 // and implements the const forward iterator concept.
00122 template <typename T>
00123 class ParamIterator {
00124  public:
00125   typedef T value_type;
00126   typedef const T& reference;

00128 
00129   // ParamIterator assumes ownership of the impl_ pointer.
00130   ParamIterator(const ParamIterator& other) : impl_(other.impl_->Clone()) {}
00131   ParamIterator& operator=(const ParamIterator& other) {
00132     if (this != &other)
00133       impl_.reset(other.impl_->Clone());
00134     return *this;
00135   }

00287 // Generates values from a pair of STL-style iterators. Used in the
00288 // ValuesIn() function. The elements are copied from the source range
00289 // since the source can be located on the stack, and the generator
00290 // is likely to persist beyond that stack frame.

00292 class ValuesInIteratorRangeGenerator : public ParamGeneratorInterface<T> {
00293  public:
00294   template <typename ForwardIterator>
00295   ValuesInIteratorRangeGenerator(ForwardIterator begin, ForwardIterator end)
00296       : container_(begin, end) {}
00297   virtual ~ValuesInIteratorRangeGenerator() {}
00298 
00299   virtual ParamIteratorInterface<T>* Begin() const {
00300     return new Iterator(this, container_.begin());
00301   }
00302   virtual ParamIteratorInterface<T>* End() const {
00303     return new Iterator(this, container_.end());
00304   }
00305 
00306  private:
00307   typedef typename ::std::vector<T> ContainerType;
00308 
00309   class Iterator : public ParamIteratorInterface<T> {
00310    public:
00311     Iterator(const ParamGeneratorInterface<T>* base,
00312              typename ContainerType::const_iterator iterator)
00313         : base_(base), iterator_(iterator) {}
00314     virtual ~Iterator() {}
00315 
00316     virtual const ParamGeneratorInterface<T>* BaseGenerator() const {
00317       return base_;
00318     }
00319     virtual void Advance() {
00320       ++iterator_;
00321       value_.reset();
00322     }
00323     virtual ParamIteratorInterface<T>* Clone() const {
00324       return new Iterator(*this);
00325     }
00326     // We need to use cached value referenced by iterator_ because *iterator_
00327     // can return a temporary object (and of type other then T), so just
00328     // having "return &*iterator_;" doesn't work.
00329     // value_ is updated here and not in Advance() because Advance()
00330     // can advance iterator_ beyond the end of the range, and we cannot
00331     // detect that fact. The client code, on the other hand, is
00332     // responsible for not calling Current() on an out-of-range iterator.
00333     virtual const T* Current() const {
00334       if (value_.get() == NULL)
00335         value_.reset(new T(*iterator_));
00336       return value_.get();
00337     }
00338     virtual bool Equals(const ParamIteratorInterface<T>& other) const {
00339       // Having the same base generator guarantees that the other
00340       // iterator is of the same type and we can downcast.
00341       GTEST_CHECK_(BaseGenerator() == other.BaseGenerator())
00342           << "The program attempted to compare iterators "
00343           << "from different generators." << std::endl;

00345           CheckedDowncastToActualType<const Iterator>(&other)->iterator_;
00346     }
00347 
00348    private:
00349     Iterator(const Iterator& other)
00350           // The explicit constructor call suppresses a false warning
00351           // emitted by gcc when supplied with the -Wextra option.

00353           base_(other.base_),
00354           iterator_(other.iterator_) {}

00356     const ParamGeneratorInterface<T>* const base_;

00363     mutable scoped_ptr<const T> value_;
00364   };  // class ValuesInIteratorRangeGenerator::Iterator

00685     for (TestCaseInfoContainer::iterator it = test_case_infos_.begin();
00686          it != test_case_infos_.end(); ++it) {
