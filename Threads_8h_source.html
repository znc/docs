00167 class CThread {

00169         typedef void *threadRoutine(void *);
00170         static void startThread(threadRoutine *func, void *arg) {
00171                 pthread_t thr;
00172                 sigset_t old_sigmask, sigmask;
00173 
00174                 /* Block all signals. The thread will inherit our signal mask
00175                  * and thus won't ever try to handle signals.
00176                  */
00177                 int i = sigfillset(&sigmask);
00178                 i |= pthread_sigmask(SIG_SETMASK, &sigmask, &old_sigmask);
00179                 i |= pthread_create(&thr, NULL, func, arg);
00180                 i |= pthread_sigmask(SIG_SETMASK, &old_sigmask, NULL);
00181                 i |= pthread_detach(thr);
00182                 if (i) {
00183                         CUtils::PrintError("Can't start new thread: "
00184                                         + CString(strerror(errno)));
00185                         exit(1);
00186                 }
00187         }
