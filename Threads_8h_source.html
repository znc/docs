179 class CThread {
180 public:
181  typedef void *threadRoutine(void *);
182  static void startThread(threadRoutine *func, void *arg) {
183  pthread_t thr;
184  sigset_t old_sigmask, sigmask;
185 
186  /* Block all signals. The thread will inherit our signal mask
187  * and thus won't ever try to handle signals.
188  */
189  int i = sigfillset(&sigmask);
190  i |= pthread_sigmask(SIG_SETMASK, &sigmask, &old_sigmask);
191  i |= pthread_create(&thr, nullptr, func, arg);
192  i |= pthread_sigmask(SIG_SETMASK, &old_sigmask, nullptr);
193  i |= pthread_detach(thr);
194  if (i) {
195  CUtils::PrintError("Can't start new thread: "
196  + CString(strerror(errno)));
197  exit(1);
198  }
199  }
200 
201 private:
202  // Undefined constructor
203  CThread();
204 };
