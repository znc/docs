00181         typedef void *threadRoutine(void *);
00182         static void startThread(threadRoutine *func, void *arg) {
00183                 pthread_t thr;
00184                 sigset_t old_sigmask, sigmask;
00185 
00186                 /* Block all signals. The thread will inherit our signal mask
00187                  * and thus won't ever try to handle signals.
00188                  */
00189                 int i = sigfillset(&sigmask);
00190                 i |= pthread_sigmask(SIG_SETMASK, &sigmask, &old_sigmask);
00191                 i |= pthread_create(&thr, NULL, func, arg);
00192                 i |= pthread_sigmask(SIG_SETMASK, &old_sigmask, NULL);
00193                 i |= pthread_detach(thr);
00194                 if (i) {
00195                         CUtils::PrintError("Can't start new thread: "
00196                                         + CString(strerror(errno)));
00197                         exit(1);
00198                 }
00199         }
