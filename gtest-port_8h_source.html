00046 // Environment-describing macros
00047 // -----------------------------
00048 //
00049 // Google Test can be used in many different environments.  Macros in
00050 // this section tell Google Test what kind of environment it is being
00051 // used in, such that Google Test can provide environment-specific
00052 // features and implementations.
00053 //
00054 // Google Test tries to automatically detect the properties of its
00055 // environment, so users usually don't need to worry about these
00056 // macros.  However, the automatic detection is not perfect.
00057 // Sometimes it's necessary for a user to define some of the following
00058 // macros in the build script to override Google Test's decisions.
00059 //
00060 // If the user doesn't define a macro in the list, Google Test will
00061 // provide a default definition.  After this header is #included, all
00062 // macros in this list will be defined to either 1 or 0.
00063 //
00064 // Notes to maintainers:
00065 //   - Each macro here is a user-tweakable knob; do not grow the list
00066 //     lightly.
00067 //   - Use #if to key off these macros.  Don't use #ifdef or "#if
00068 //     defined(...)", which will not work as these macros are ALWAYS
00069 //     defined.
00070 //
00071 //   GTEST_HAS_CLONE          - Define it to 1/0 to indicate that clone(2)
00072 //                              is/isn't available.
00073 //   GTEST_HAS_EXCEPTIONS     - Define it to 1/0 to indicate that exceptions
00074 //                              are enabled.
00075 //   GTEST_HAS_GLOBAL_STRING  - Define it to 1/0 to indicate that ::string
00076 //                              is/isn't available (some systems define
00077 //                              ::string, which is different to std::string).
00078 //   GTEST_HAS_GLOBAL_WSTRING - Define it to 1/0 to indicate that ::string
00079 //                              is/isn't available (some systems define
00080 //                              ::wstring, which is different to std::wstring).
00081 //   GTEST_HAS_POSIX_RE       - Define it to 1/0 to indicate that POSIX regular
00082 //                              expressions are/aren't available.
00083 //   GTEST_HAS_PTHREAD        - Define it to 1/0 to indicate that <pthread.h>
00084 //                              is/isn't available.
00085 //   GTEST_HAS_RTTI           - Define it to 1/0 to indicate that RTTI is/isn't
00086 //                              enabled.
00087 //   GTEST_HAS_STD_WSTRING    - Define it to 1/0 to indicate that
00088 //                              std::wstring does/doesn't work (Google Test can
00089 //                              be used where std::wstring is unavailable).
00090 //   GTEST_HAS_TR1_TUPLE      - Define it to 1/0 to indicate tr1::tuple
00091 //                              is/isn't available.
00092 //   GTEST_HAS_SEH            - Define it to 1/0 to indicate whether the
00093 //                              compiler supports Microsoft's "Structured
00094 //                              Exception Handling".
00095 //   GTEST_HAS_STREAM_REDIRECTION
00096 //                            - Define it to 1/0 to indicate whether the
00097 //                              platform supports I/O stream redirection using
00098 //                              dup() and dup2().
00099 //   GTEST_USE_OWN_TR1_TUPLE  - Define it to 1/0 to indicate whether Google
00100 //                              Test's own tr1 tuple implementation should be
00101 //                              used.  Unused when the user sets
00102 //                              GTEST_HAS_TR1_TUPLE to 0.
00103 //   GTEST_LANG_CXX11         - Define it to 1/0 to indicate that Google Test
00104 //                              is building in C++11/C++98 mode.
00105 //   GTEST_LINKED_AS_SHARED_LIBRARY
00106 //                            - Define to 1 when compiling tests that use
00107 //                              Google Test as a shared library (known as
00108 //                              DLL on Windows).
00109 //   GTEST_CREATE_SHARED_LIBRARY
00110 //                            - Define to 1 when compiling Google Test itself
00111 //                              as a shared library.

01009 // The GTEST_COMPILE_ASSERT_ macro can be used to verify that a compile time
01010 // expression is true. For example, you could use it to verify the
01011 // size of a static array:
01012 //
01013 //   GTEST_COMPILE_ASSERT_(GTEST_ARRAY_SIZE_(names) == NUM_NAMES,
01014 //                         names_incorrect_size);
01015 //
01016 // or to make sure a struct is smaller than a certain size:
01017 //
01018 //   GTEST_COMPILE_ASSERT_(sizeof(foo) < 128, foo_too_large);
01019 //
01020 // The second argument to the macro is the name of the variable. If
01021 // the expression is false, most compilers will issue a warning/error
01022 // containing the name of the variable.
01023 
01024 #if GTEST_LANG_CXX11
01025 # define GTEST_COMPILE_ASSERT_(expr, msg) static_assert(expr, #msg)
01026 #else  // !GTEST_LANG_CXX11
01027 template <bool>
01028   struct CompileAssert {
01029 };
01030 
01031 # define GTEST_COMPILE_ASSERT_(expr, msg) \
01032   typedef ::testing::internal::CompileAssert<(static_cast<bool>(expr))> \
01033       msg[static_cast<bool>(expr) ? 1 : -1] GTEST_ATTRIBUTE_UNUSED_
01034 #endif  // !GTEST_LANG_CXX11
01035 
01036 // Implementation details of GTEST_COMPILE_ASSERT_:
01037 //
01038 // (In C++11, we simply use static_assert instead of the following)
01039 //
01040 // - GTEST_COMPILE_ASSERT_ works by defining an array type that has -1
01041 //   elements (and thus is invalid) when the expression is false.
01042 //
01043 // - The simpler definition
01044 //
01045 //    #define GTEST_COMPILE_ASSERT_(expr, msg) typedef char msg[(expr) ? 1 : -1]
01046 //
01047 //   does not work, as gcc supports variable-length arrays whose sizes
01048 //   are determined at run-time (this is gcc's extension and not part
01049 //   of the C++ standard).  As a result, gcc fails to reject the
01050 //   following code with the simple definition:
01051 //
01052 //     int foo;
01053 //     GTEST_COMPILE_ASSERT_(foo, msg); // not supposed to compile as foo is
01054 //                                      // not a compile-time constant.
01055 //
01056 // - By using the type CompileAssert<(bool(expr))>, we ensures that
01057 //   expr is a compile-time constant.  (Template arguments must be
01058 //   determined at compile-time.)
01059 //
01060 // - The outter parentheses in CompileAssert<(bool(expr))> are necessary
01061 //   to work around a bug in gcc 3.4.4 and 4.0.1.  If we had written
01062 //
01063 //     CompileAssert<bool(expr)>
01064 //
01065 //   instead, these compilers will refuse to compile
01066 //

01321 // INTERNAL IMPLEMENTATION - DO NOT USE IN USER CODE.
01322 //
01323 // Use ImplicitCast_ as a safe version of static_cast for upcasting in
01324 // the type hierarchy (e.g. casting a Foo* to a SuperclassOfFoo* or a
01325 // const Foo*).  When you use ImplicitCast_, the compiler checks that
01326 // the cast is safe.  Such explicit ImplicitCast_s are necessary in
01327 // surprisingly many situations where C++ demands an exact type match
01328 // instead of an argument type convertable to a target type.
01329 //
01330 // The syntax for using ImplicitCast_ is the same as for static_cast:
01331 //

01345 // SuperclassOfFoo), it's fine to use ImplicitCast_<>, since upcasts
01346 // always succeed.  When you downcast (that is, cast a pointer from
01347 // type Foo to type SubclassOfFoo), static_cast<> isn't safe, because
01348 // how do you know the pointer is really of type SubclassOfFoo?  It
01349 // could be a bare Foo, or of type DifferentSubclassOfFoo.  Thus,
01350 // when you downcast, you should use this macro.  In debug mode, we
01351 // use dynamic_cast<> to double-check the downcast is legal (we die

01354 // the cast is legal!

01356 // In particular, you SHOULDN'T be using dynamic_cast<> in order to
01357 // do RTTI (eg code like this:
01358 //    if (dynamic_cast<Subclass1>(foo)) HandleASubclass1Object(foo);

01374     const To to = NULL;

01376   }
