00232 // This template class represents an IEEE floating-point number
00233 // (either single-precision or double-precision, depending on the
00234 // template parameters).
00235 //
00236 // The purpose of this class is to do more sophisticated number
00237 // comparison.  (Due to round-off error, etc, it's very unlikely that
00238 // two floating-points will be equal exactly.  Hence a naive
00239 // comparison by the == operation often doesn't work.)
00240 //
00241 // Format of IEEE floating-point:
00242 //
00243 //   The most-significant bit being the leftmost, an IEEE
00244 //   floating-point looks like
00245 //
00246 //     sign_bit exponent_bits fraction_bits
00247 //
00248 //   Here, sign_bit is a single bit that designates the sign of the
00249 //   number.
00250 //
00251 //   For float, there are 8 exponent bits and 23 fraction bits.
00252 //
00253 //   For double, there are 11 exponent bits and 52 fraction bits.
00254 //
00255 //   More details can be found at
00256 //   http://en.wikipedia.org/wiki/IEEE_floating-point_standard.
00257 //
00258 // Template parameter:
00259 //
00260 //   RawType: the raw floating-point type (either float or double)
00261 template <typename RawType>
00262 class FloatingPoint {
00263  public:
00264   // Defines the unsigned integer type that has the same size as the
00265   // floating point number.
00266   typedef typename TypeWithSize<sizeof(RawType)>::UInt Bits;
00267 
00268   // Constants.
00269 
00270   // # of bits in a number.
00271   static const size_t kBitCount = 8*sizeof(RawType);
00272 
00273   // # of fraction bits in a number.
00274   static const size_t kFractionBitCount =
00275     std::numeric_limits<RawType>::digits - 1;
00276 
00277   // # of exponent bits in a number.
00278   static const size_t kExponentBitCount = kBitCount - 1 - kFractionBitCount;
00279 
00280   // The mask for the sign bit.
00281   static const Bits kSignBitMask = static_cast<Bits>(1) << (kBitCount - 1);
00282 
00283   // The mask for the fraction bits.
00284   static const Bits kFractionBitMask =
00285     ~static_cast<Bits>(0) >> (kExponentBitCount + 1);
00286 
00287   // The mask for the exponent bits.
00288   static const Bits kExponentBitMask = ~(kSignBitMask | kFractionBitMask);
00289 
00290   // How many ULP's (Units in the Last Place) we want to tolerate when
00291   // comparing two numbers.  The larger the value, the more error we
00292   // allow.  A 0 value means that two numbers must be exactly the same
00293   // to be considered equal.
00294   //
00295   // The maximum error of a single floating-point operation is 0.5
00296   // units in the last place.  On Intel CPU's, all floating-point
00297   // calculations are done with 80-bit precision, while double has 64
00298   // bits.  Therefore, 4 should be enough for ordinary use.
00299   //
00300   // See the following article for more details on ULP:
00301   // http://randomascii.wordpress.com/2012/02/25/comparing-floating-point-numbers-2012-edition/
00302   static const size_t kMaxUlps = 4;
00303 
00304   // Constructs a FloatingPoint from a raw floating-point number.
00305   //
00306   // On an Intel CPU, passing a non-normalized NAN (Not a Number)
00307   // around may change its bits, although the new value is guaranteed
00308   // to be also a NAN.  Therefore, don't expect this constructor to
00309   // preserve the bits in x when x is a NAN.
00310   explicit FloatingPoint(const RawType& x) { u_.value_ = x; }
00311 
00312   // Static methods
00313 
00314   // Reinterprets a bit pattern as a floating-point number.
00315   //
00316   // This function is needed to test the AlmostEquals() method.
00317   static RawType ReinterpretBits(const Bits bits) {
00318     FloatingPoint fp(0);
00319     fp.u_.bits_ = bits;
00320     return fp.u_.value_;
00321   }
00322 
00323   // Returns the floating-point number that represent positive infinity.
00324   static RawType Infinity() {
00325     return ReinterpretBits(kExponentBitMask);
00326   }
00327 
00328   // Returns the maximum representable finite floating-point number.
00329   static RawType Max();
00330 
00331   // Non-static methods
00332 
00333   // Returns the bits that represents this number.
00334   const Bits &bits() const { return u_.bits_; }
00335 
00336   // Returns the exponent bits of this number.
00337   Bits exponent_bits() const { return kExponentBitMask & u_.bits_; }
00338 
00339   // Returns the fraction bits of this number.
00340   Bits fraction_bits() const { return kFractionBitMask & u_.bits_; }
00341 
00342   // Returns the sign bit of this number.
00343   Bits sign_bit() const { return kSignBitMask & u_.bits_; }
00344 
00345   // Returns true iff this is NAN (not a number).
00346   bool is_nan() const {
00347     // It's a NAN if the exponent bits are all ones and the fraction
00348     // bits are not entirely zeros.
00349     return (exponent_bits() == kExponentBitMask) && (fraction_bits() != 0);
00350   }
00351 
00352   // Returns true iff this number is at most kMaxUlps ULP's away from
00353   // rhs.  In particular, this function:
00354   //
00355   //   - returns false if either number is (or both are) NAN.
00356   //   - treats really large numbers as almost equal to infinity.
00357   //   - thinks +0.0 and -0.0 are 0 DLP's apart.
00358   bool AlmostEquals(const FloatingPoint& rhs) const {
00359     // The IEEE standard says that any comparison operation involving
00360     // a NAN must return false.
00361     if (is_nan() || rhs.is_nan()) return false;
00362 
00363     return DistanceBetweenSignAndMagnitudeNumbers(u_.bits_, rhs.u_.bits_)
00364         <= kMaxUlps;
00365   }
00366 
00367  private:
00368   // The data type used to store the actual floating-point number.
00369   union FloatingPointUnion {
00370     RawType value_;  // The raw floating-point number.
00371     Bits bits_;      // The bits that represent the number.
00372   };
00373 
00374   // Converts an integer from the sign-and-magnitude representation to
00375   // the biased representation.  More precisely, let N be 2 to the
00376   // power of (kBitCount - 1), an integer x is represented by the
00377   // unsigned number x + N.
00378   //
00379   // For instance,
00380   //
00381   //   -N + 1 (the most negative number representable using
00382   //          sign-and-magnitude) is represented by 1;
00383   //   0      is represented by N; and
00384   //   N - 1  (the biggest number representable using
00385   //          sign-and-magnitude) is represented by 2N - 1.
00386   //
00387   // Read http://en.wikipedia.org/wiki/Signed_number_representations
00388   // for more details on signed number representations.
00389   static Bits SignAndMagnitudeToBiased(const Bits &sam) {
00390     if (kSignBitMask & sam) {
00391       // sam represents a negative number.
00392       return ~sam + 1;
00393     } else {
00394       // sam represents a positive number.
00395       return kSignBitMask | sam;
00396     }
00397   }
00398 
00399   // Given two numbers in the sign-and-magnitude representation,
00400   // returns the distance between them as an unsigned number.
00401   static Bits DistanceBetweenSignAndMagnitudeNumbers(const Bits &sam1,
00402                                                      const Bits &sam2) {
00403     const Bits biased1 = SignAndMagnitudeToBiased(sam1);
00404     const Bits biased2 = SignAndMagnitudeToBiased(sam2);
00405     return (biased1 >= biased2) ? (biased1 - biased2) : (biased2 - biased1);
00406   }
00407 
00408   FloatingPointUnion u_;
00409 };

00851 // ImplicitlyConvertible<From, To>::value is a compile-time bool
00852 // constant that's true iff type From can be implicitly converted to
00853 // type To.
00854 template <typename From, typename To>
00855 class ImplicitlyConvertible {
00856  private:
00857   // We need the following helper functions only for their types.
00858   // They have no implementations.
00859 
00860   // MakeFrom() is an expression whose type is From.  We cannot simply
00861   // use From(), as the type From may not have a public default
00862   // constructor.
00863   static typename AddReference<From>::type MakeFrom();
00864 
00865   // These two functions are overloaded.  Given an expression
00866   // Helper(x), the compiler will pick the first version if x can be
00867   // implicitly converted to type To; otherwise it will pick the
00868   // second version.
00869   //
00870   // The first version returns a value of size 1, and the second
00871   // version returns a value of size 2.  Therefore, by checking the
00872   // size of Helper(x), which can be done at compile time, we can tell
00873   // which version of Helper() is used, and hence whether x can be
00874   // implicitly converted to type To.
00875   static char Helper(To);
00876   static char (&Helper(...))[2];  // NOLINT

00878   // We have to put the 'public' section after the 'private' section,
00879   // or MSVC refuses to compile the code.
00880  public:
00881 #if defined(__BORLANDC__)
00882   // C++Builder cannot use member overload resolution during template
00883   // instantiation.  The simplest workaround is to use its C++0x type traits
00884   // functions (C++Builder 2009 and above only).
00885   static const bool value = __is_convertible(From, To);
00886 #else
00887   // MSVC warns about implicitly converting from double to int for
00888   // possible loss of data, so we need to temporarily disable the
00889   // warning.
00890   GTEST_DISABLE_MSC_WARNINGS_PUSH_(4244)
00891   static const bool value =
00892       sizeof(Helper(ImplicitlyConvertible::MakeFrom())) == 1;
00893   GTEST_DISABLE_MSC_WARNINGS_POP_()
00894 #endif  // __BORLANDC__
00895 };
